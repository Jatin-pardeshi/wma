    let str = document.getElementById("inputStr").value.toLowerCase();
    let vowels = "";

    for (let i = 0; i < str.length; i++) {
        if ("aeiou".includes(str[i])) {

Q2. Write a Kotlin program to calculate the factorial of a given number. (Use Function)

fun factorial(num: Int): Long {
    var result: Long = 1
    for (i in 1..num) {
        result *= i
    }
    return result
}

fun main() {
    print("Enter a number: ")
    val input = readLine()?.toIntOrNull()

    if (input != null && input >= 0) {
        val fact = factorial(input)
        println("Factorial of $input is: $fact")
    } else {
        println("Please enter a valid non-negative integer.")
    }
}

Q2. Write a Kotlin application to check whether given number is prime or not (Use Function)


fun isPrime(number: Int): Boolean {
    if (number <= 1) return false
    for (i in 2..number / 2) {
        if (number % i == 0) {
            return false
        }
    }
    return true
}

fun main() {
    print("Enter a number: ")
    val input = readLine()?.toIntOrNull()

    if (input != null) {
        if (isPrime(input)) {
            println("$input is a Prime number.")
        } else {
            println("$input is NOT a Prime number.")
        }
    } else {
        println("Please enter a valid integer.")
    }
}

Q2. Write a PHP code to check whether given number is Armstrong or not.



<!DOCTYPE html>
<html>
<head>
    <title>Armstrong Number Check</title>
</head>
<body>
    <h2>Check if a Number is Armstrong</h2>
    <form method="POST">
        <label>Enter a number: </label>
        <input type="number" name="number" required>
        <input type="submit" value="Check">
    </form>

    <?php
    if ($_SERVER["REQUEST_METHOD"] == "POST") {
        // Get the number from the form
        $number = $_POST['number'];
        $sum = 0;
        $temp = $number;
        // Calculate the number of digits in the number
        $numDigits = strlen($number);
        
        // Process each digit
        while ($temp > 0) {
            $digit = $temp % 10;
            // Raise the digit to the power of the number of digits
            $sum += pow($digit, $numDigits);
            $temp = floor($temp / 10);
        }
        
        // Display the result based on the Armstrong condition
        if ($sum == $number) {
            echo "<p>$number is an Armstrong number.</p>";
        } else {
            echo "<p>$number is not an Armstrong number.</p>";
        }
    }
    ?>
</body>

Q1. Write a javascript code to calculate the sum of first and last digit of a number (Handle
onBlur event)

<!DOCTYPE html>
<html>
<head>
  <title>First and Last Digit Sum</title>
  <script>
    function sumFirstLast() {
      let numStr = document.getElementById("number").value.trim();
      let result = document.getElementById("result");

      if (/^\d+$/.test(numStr)) {
        let firstDigit = parseInt(numStr.charAt(0));
        let lastDigit = parseInt(numStr.charAt(numStr.length - 1));
        let sum = firstDigit + lastDigit;

        result.textContent = "Sum of first and last digit: " + sum;
      } else {
        result.textContent = "Please enter a valid number";
        result.style.color = "red";
      }
    }
  </script>
</head>
<body>
  <h2>Calculate Sum of First and Last Digit</h2>
  <label>Enter a number:</label>
  <input type="text" id="number" onblur="sumFirstLast()">
  <p id="result"></p>
</body>
</html>

Q1. Write a JavaScript code to validate Pin Code. (Handle OnBlur event) [15 Marks]


<!DOCTYPE html>
<html>
<head>
  <title>Pin Code Validation</title>
  <script>
    function validatePin() {
      const pin = document.getElementById("pin").value;
      const message = document.getElementById("pinMessage");

      if (/^\d{6}$/.test(pin)) {
        message.textContent = "Valid Pin Code";
        message.style.color = "green";
      } else {
        message.textContent = "Invalid Pin Code (Must be 6 digits)";
        message.style.color = "red";
      }
    }
  </script>
</head>
<body>
  <h2>Pin Code Validation</h2>
  <label>Enter Pin Code:</label>
  <input type="text" id="pin" onblur="validatePin()">
  <p id="pinMessage"></p>
</body>
